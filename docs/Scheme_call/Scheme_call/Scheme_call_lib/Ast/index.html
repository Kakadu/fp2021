type datum =
| DConst of dconst
| DList of datum list
| DAbreviation of char * datum
and id = string
and dconst =
| DInt of int
| DBool of bool
| DString of string
| DSymbol of id
and variable = id
and const =
| Int of int
| Bool of bool
| String of string
and operator =
| Op of expr
and formals =
| FVarList of variable list
| FVar of variable
and expr =
| Var of variable
| Quote of datum
| Const of const
| ProcCall of operator * expr list
| Lam of formals * definition list * expr * expr list
| Cond of expr * expr * expr option
and definition = variable * expr
and form =
| Def of definition
| Expr of expr
and program = form list
